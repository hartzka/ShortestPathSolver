31      /**
32       * The main logic in A* algorithm. Checks if currentNode's neighbour is
33       * found in closed set and open list. Updates node's information and checks
34       * if a better path exists from currentNode to its neighbour.
35       *
36       * @param currentNode Current node
37       * @param col Neighbour's column
38       * @param row Neighbour's row
39       * @param cost Movement cost
40       */
41      private void checkNode(Node currentNode, int col, int row, int cost) {
42          Node neighbour = sr.getNodes()[row][col];
43          if (!neighbour.isBlock()) {
44              if (!closedSet.contains(neighbour) && !openList.contains(neighbour)) {
45                  neighbour.setAStarInformation(currentNode, cost);
46                  openList.add(neighbour);
47              } else if (closedSet.contains(neighbour)) { // in case the algorithm doesn't work. Otherwise nodes in closed set shouldn't be processed.
48                  if (checkifBetterPathExists(currentNode, neighbour, cost)) {
49                      openList.add(neighbour);
50                  }
51              } else {
52                  checkifBetterPathExists(currentNode, neighbour, cost);
53              }
54          }
55      }

57      /**
58       * The main method, which searches the shortest route with either A* or JPS
59       *
60       * @param initialNode Beginning of the search
61       * @return List presenting the route
62       */
63      @Override
64      public CustomArrayList<Node> calculatePath(Node initialNode) {
65          openList.add(initialNode);
66          while (openList.size() != 0) {
67              Node currentNode = (Node) openList.poll(); // The node with smallest distance (f-value)
68              closedSet.add(currentNode);
69              if (sr.isFinalNode(currentNode)) {
70                  //System.out.println("G: " + currentNode.getG());
71                  return getPath(currentNode);
72              } else {
73                  if (!jps) {
74                      addNeighbourNodes(currentNode); //A*-algorithm
75                  } else { //A* with JPS
76                      Node[] jumpPoints = calculateJumpPoints(currentNode); //Possible jump points in JPS
77                      for (int i = 0; i < jumpPoints.length; i++) {
78                          Node jumpPoint = jumpPoints[i];
79                          if (!(jumpPoint == null || jumpPoint.isBlock())) {
80                              if (!closedSet.contains(jumpPoint) && !openList.contains(jumpPoint)) { // Logic in JPS
81                                  setAStarInformationRange(currentNode, jumpPoint);
82                                  openList.add(jumpPoint);
83                              } else if (closedSet.contains(jumpPoint)) {
84                                  if (checkifBetterPathExists(currentNode, jumpPoint, approxG(currentNode.getColumn(), currentNode.getRow(), jumpPoint.getColumn(), jumpPoint.getRow()))) {
85                                      openList.add(jumpPoint);
86                                  }
87                              } else {
88                                  checkifBetterPathExists(currentNode, jumpPoint, approxG(currentNode.getColumn(), currentNode.getRow(), jumpPoint.getColumn(), jumpPoint.getRow()));
89                              }
90                          }
91                      }
92                  }
93              }
94          }
95          return new CustomArrayList<>();
96      }

98      /**
99       * Returns node's jumppoints for JPS. This speeds up the search remarkably.
100      *
101      * @param node
102      * @return possible jump points
103      */
104     public Node[] calculateJumpPoints(Node node) {
105         Node[] jumpPoints = new Node[8];
106         Pair[] neighbors = getNeighbours(node); // All neighbours
107         for (int i = 0; i < neighbors.length; i++) {
108 
109             Pair<Integer, Integer> point = jump((int) neighbors[i].getKey(), (int) neighbors[i].getValue(), node.getColumn(), node.getRow()); //Pair presenting jump point
110             if (point.getKey() != -1) {
111                 int x = point.getKey();
112                 int y = point.getValue();
113 
114                 int newG = (approxG(x, y, node.getColumn(), node.getRow()));
115                 if (sr.getNodes()[y][x].getDist() == 0 || sr.getNodes()[y][x].getG() > newG) {
116                     jumpPoints[i] = sr.getNodes()[y][x]; // Jump point is added
117                 }
118             }
119         }
120         return jumpPoints;
121     }

123     /**
124      * Recursive method, which searches a jump point based on a node and its
125      * parent.
126      *
127      * @param x node's column
128      * @param y node's row
129      * @param px parent's column
130      * @param py parent's row
131      * @return coordinates of the possibly found jump point
132      */
133     public Pair jump(int x, int y, int px, int py) {
134 
135         int dx = (x - px) / Math.max(Math.abs(x - px), 1);
136         int dy = (y - py) / Math.max(Math.abs(y - py), 1);
137 
138         if (!allowed(x, y)) {
139             return new Pair(-1, -1);
140         }
141         if (x == sr.getFinalNode().getColumn() && y == sr.getFinalNode().getRow()) {
142             return new Pair(x, y);
143         }
144         if (dx != 0 && dy != 0) {
145             if ((allowed(x - dx, y + dy) && !allowed(x - dx, y)) || (allowed(x + dx, y - dy) && !allowed(x, y - dy))) {
146                 return new Pair(x, y);
147             }
148             Pair jumpx = jump(x + dx, y, x, y);
149             Pair jumpy = jump(x, y + dy, x, y);
150             if ((int) jumpx.getKey() != -1 || (int) jumpy.getKey() != -1) {
151                 return new Pair(x, y);
152             }
153         } else {
154             if (dx == 0) {
155                 if ((allowed(x + 1, y + dy) && !allowed(x + 1, y)) || (allowed(x - 1, y + dy) && !allowed(x - 1, y))) {
156                     return new Pair(x, y);
157                 }
158             } else {
159                 if ((allowed(x + dx, y + 1) && !allowed(x, y + 1)) || (allowed(x + dx, y - 1) && !allowed(x, y - 1))) {
160                     return new Pair(x, y);
161                 }
162             }
163         }
164 
165         if (x == 0 && y == 0 && dx == 0 && dy == 0) {
166             return new Pair(-1, -1);
167         }
168         Pair table = jump(x + dx, y + dy, x, y);
169         return table;
170     }

172     /**
173      * Returns all allowed node's neighbours
174      *
175      * @param node
176      * @return table presenting neighbours
177      */
178     @Override
179     public Pair[] getNeighbours(Node node) {
180         Pair[] neighbors = new Pair[8];
181         int x = node.getColumn();
182         int y = node.getRow();
183 
184         neighbors[0] = allowed(x, y - 1) ? new Pair(x, y - 1) : new Pair(-1, -1);
185         neighbors[1] = allowed(x + 1, y) ? new Pair(x + 1, y) : new Pair(-1, -1);
186         neighbors[2] = allowed(x, y + 1) ? new Pair(x, y + 1) : new Pair(-1, -1);
187         neighbors[3] = allowed(x - 1, y) ? new Pair(x - 1, y) : new Pair(-1, -1);
188         neighbors[4] = allowed(x - 1, y - 1) ? new Pair(x - 1, y - 1) : new Pair(-1, -1);
189         neighbors[5] = allowed(x + 1, y - 1) ? new Pair(x + 1, y - 1) : new Pair(-1, -1);
190         neighbors[6] = allowed(x + 1, y + 1) ? new Pair(x + 1, y + 1) : new Pair(-1, -1);
191         neighbors[7] = allowed(x - 1, y + 1) ? new Pair(x - 1, y + 1) : new Pair(-1, -1);
192 
193         return neighbors;
194     }

196     private void addNeighbourNodes(Node currentNode) {
197         int row = currentNode.getRow();
198         int col = currentNode.getColumn();
199         if (row - 1 >= 0) { //ylin rivi
200             checkNode(currentNode, col, row - 1, hCost); //upper
201             if (col - 1 >= 0) {
202                 checkNode(currentNode, col - 1, row - 1, diagonalCost); //upper left
203             }
204             if (col + 1 < sr.getCols()) {
205                 checkNode(currentNode, col + 1, row - 1, diagonalCost); //upper right
206             }
207         }
208         if (col - 1 >= 0) {
209             checkNode(currentNode, col - 1, row, hCost); //left
210         }
211         if (col + 1 < sr.getCols()) {
212             checkNode(currentNode, col + 1, row, hCost); //right
213         }
214         if (row + 1 < sr.getRows()) { //lower row
215             checkNode(currentNode, col, row + 1, hCost); //bottom
216             if (col - 1 >= 0) {
217                 checkNode(currentNode, col - 1, row + 1, diagonalCost); //bottom left
218             }
219             if (col + 1 < sr.getCols()) {
220                 checkNode(currentNode, col + 1, row + 1, diagonalCost); //bottom right
221             }
222         }
223     }

244     /**
245      *
246      * Calculates a heuristic from current to finalNode. The pattern may need
247      * updating depending on input. The heuristic may yield wrong results if not
248      * defined correctly.
249      *
250      * @param current Current node
251      * @param finalNode Final node
252      */
253     public void calculateHeuristic(Node current, Node finalNode) {
254         int h = (int) Math.sqrt(Math.pow(finalNode.getColumn() - current.getColumn(), 3) + Math.pow(finalNode.getRow() - current.getRow(), 3));
255         int euclidian = (int) Math.sqrt(Math.pow(finalNode.getColumn() - current.getColumn(), 2) + Math.pow(finalNode.getRow() - current.getRow(), 2));
256         int manhattan = (int) (Math.abs(finalNode.getColumn() - current.getColumn()) + Math.abs(finalNode.getRow() - current.getRow()));
257 
258         current.setH(euclidian);
259     }

261     /**
262      *
263      * Checks if a better path exists between currentNode and neighbour, and if
264      * exists, updates neighbour's information.
265      *
266      * @param currentNode Current node
267      * @param neighbour Neighbour node
268      * @param cost Movement cost
269      * @return true if a better path exists, false otherwise
270      */
271     public boolean checkifBetterPathExists(Node currentNode, Node neighbour, int cost) {
272         int gCost = currentNode.getG() + cost;
273         if (gCost < neighbour.getG()) {
274             neighbour.setAStarInformation(currentNode, cost);
275             return true;
276         }
277         return false;
278     }

280     /**
281      * Gives an approximation of g value between two nodes.
282      *
283      * @param x1 x-coordinate of the first node
284      * @param y1 y-coordinate of the first node
285      * @param x2 x-coordinate of the second node
286      * @param y2 y-coordinate of the second node
287      *
288      * @return Approximation of g cost
289      */
290     private int approxG(int x1, int y1, int x2, int y2) {
291         if (x1 == x2) {
292             return Math.abs(y1 - y2) * hCost;
293         } else if (y1 == y2) {
294             return Math.abs(x1 - x2) * hCost;
295         } else {
296             return Math.abs(x1 - x2) * diagonalCost;
297         }
298     }

304     /**
305      * Updates all Node informations in a range between currentNode and
306      * jumpPoint. Used in JPS.
307      *
308      * @param currentNode Current Node
309      * @param jumpPoint Jump point
310      *
311      */
312     private void setAStarInformationRange(Node currentNode, Node jumpPoint) {
313         int col1 = currentNode.getColumn();
314         int col2 = jumpPoint.getColumn();
315         int row1 = currentNode.getRow();
316         int row2 = jumpPoint.getRow();
317         int i = 0, j = 0;
318         if (col2 < col1) {
319             i = -1;
320         } else if (col1 < col2) {
321             i = 1;
322         }
323         if (row2 < row1) {
324             j = -1;
325         } else if (row1 < row2) {
326             j = 1;
327         }
328 
329         int a = col1, b = row1;
330         while (true) {
331             a += i;
332             b += j;
333 
334             Node n = sr.getNodes()[b][a];
335             int newG = approxG(currentNode.getColumn(), currentNode.getRow(), n.getColumn(), n.getRow());
336             if (n.getG() == 0 || currentNode.getG() + newG < n.getG()) {
337                 if (!sr.isInitialNode(n)) {
338                     sr.getNodes()[b][a].setAStarInformation(currentNode, newG);
339                 }
340             }
341 
342             if (a == col2 && b == row2) {
343                 break;
344             }
345         }
346     }
