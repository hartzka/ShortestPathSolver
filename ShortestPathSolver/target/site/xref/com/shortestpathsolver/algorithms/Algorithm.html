1   package com.shortestpathsolver.algorithms;
2   
3   import com.shortestpathsolver.domain.Node;
4   import com.shortestpathsolver.domain.ShortestRoute;
5   import com.shortestpathsolver.structures.CustomArrayList;
6   import com.shortestpathsolver.structures.Pair;
7   
8   /**
9    * Abstract algorithm class. Contains implemented and unimplemented methods used
10   * in algorithm classes.
11   *
12   * @author kaihartz
13   */
14  public abstract class Algorithm {
15  
16      private ShortestRoute sr;
17      private CustomArrayList<Node> finalPath;
18  
19      public Algorithm(ShortestRoute sr) {
20          this.sr = sr;
21          this.finalPath = new CustomArrayList<>();
22      }
23  
24      /**
25       * The main logic, which searches the shortest route
26       *
27       * @param initialNode Beginning of the search
28       * @return List presenting the route
29       */
30      public abstract CustomArrayList<Node> calculatePath(Node initialNode);
31  
32      /**
33       * Returns all allowed node's neighbours
34       *
35       * @param node
36       * @return table presenting neighbours
37       */
38      public Pair<Node, Integer>[] getNeighbours(Node node) {
39          Pair<Node, Integer>[] neighbors = new Pair[8];
40          int x = node.getColumn();
41          int y = node.getRow();
42          Node[][] nodes = sr.getNodes();
43  
44          neighbors[0] = allowed(x, y - 1) ? new Pair(nodes[y - 1][x], 10) : null;
45          neighbors[1] = allowed(x + 1, y) ? new Pair(nodes[y][x + 1], 10) : null;
46          neighbors[2] = allowed(x, y + 1) ? new Pair(nodes[y + 1][x], 10) : null;
47          neighbors[3] = allowed(x - 1, y) ? new Pair(nodes[y][x - 1], 10) : null;
48          neighbors[4] = allowed(x - 1, y - 1) ? new Pair(nodes[y - 1][x - 1], 14) : null;
49          neighbors[5] = allowed(x + 1, y - 1) ? new Pair(nodes[y - 1][x + 1], 14) : null;
50          neighbors[6] = allowed(x + 1, y + 1) ? new Pair(nodes[y + 1][x + 1], 14) : null;
51          neighbors[7] = allowed(x - 1, y + 1) ? new Pair(nodes[y + 1][x - 1], 14) : null;
52  
53          return neighbors;
54      }
55  
56      public CustomArrayList<Node> getPath(Node currentNode) {
57          CustomArrayList<Node> path = new CustomArrayList<>();
58          path.add(currentNode);
59          Node parent;
60          while ((parent = currentNode.getParent()) != null) {
61              path.add(0, parent);
62              currentNode = parent;
63          }
64          this.finalPath = path;
65          return path;
66      }
67  
68      public boolean allowed(int x, int y) {
69          return (x < sr.getCols() && y < sr.getRows() && x >= 0 && y >= 0 && sr.getBlocks()[y][x] == false);
70      }
71  
72      public abstract CustomArrayList<Node> getClosedSet();
73  
74      public CustomArrayList<Node> getPath() {
75          return this.finalPath;
76      }
77  }
